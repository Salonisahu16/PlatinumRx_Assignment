--query1:Find the revenue we got from each sales channel in a given year
SELECT 
    sales_channel,
    SUM(amount) AS total_revenue
FROM clinic_sales
WHERE strftime('%Y', datetime) = '2021'
GROUP BY sales_channel;

--query2:Find top 10 the most valuable customers for a given yearbooking_commercials
SELECT 
    uid,
    SUM(amount) AS total_spent
FROM clinic_sales
WHERE strftime('%Y', datetime) = '2021'
GROUP BY uid
ORDER BY total_spent DESC
LIMIT 10;

--query3:Find month wise revenue, expense, profit , status (profitable / not-profitable) for a given yearbooking_commercials
SELECT 
    strftime('%m', cs.datetime) AS month,
    
    SUM(cs.amount) AS total_revenue,
    
    (
        SELECT SUM(e.amount)
        FROM expenses e
        WHERE strftime('%m', e.datetime) = strftime('%m', cs.datetime)
    ) AS total_expense,
    
    SUM(cs.amount) - (
        SELECT SUM(e.amount)
        FROM expenses e
        WHERE strftime('%m', e.datetime) = strftime('%m', cs.datetime)
    ) AS profit,
    
    CASE 
        WHEN (SUM(cs.amount) - (
            SELECT SUM(e.amount)
            FROM expenses e
            WHERE strftime('%m', e.datetime) = strftime('%m', cs.datetime)
        )) > 0 
        THEN 'Profitable'
        ELSE 'Loss'
    END AS status

FROM clinic_sales cs
GROUP BY month;

--query4:For each city find the most profitable clinic for a given month
SELECT city, cid, profit
FROM (
    SELECT 
        c.city,
        c.cid,
        (SUM(cs.amount) - SUM(e.amount)) AS profit
    FROM clinics c
    JOIN clinic_sales cs ON c.cid = cs.cid
    JOIN expenses e ON c.cid = e.cid
    GROUP BY c.city, c.cid
) t
WHERE profit = (
    SELECT MAX(profit)
    FROM (
        SELECT 
            c2.city,
            c2.cid,
            (SUM(cs2.amount) - SUM(e2.amount)) AS profit
        FROM clinics c2
        JOIN clinic_sales cs2 ON c2.cid = cs2.cid
        JOIN expenses e2 ON c2.cid = e2.cid
        GROUP BY c2.city, c2.cid
    ) t2
    WHERE t2.city = t.city
);

--query5: For each state find the second least profitable clinic for a given month
SELECT *
FROM (
    SELECT c.state, c.cid,
           (SUM(cs.amount) - SUM(e.amount)) AS profit
    FROM clinics c
    JOIN clinic_sales cs ON c.cid = cs.cid
    JOIN expenses e ON c.cid = e.cid
    GROUP BY c.state, c.cid
    ORDER BY c.state, profit ASC
)
GROUP BY state
LIMIT 2;
